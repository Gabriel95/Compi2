/*
java -jar jflex-1.6.1.jar lexer

terminal PLUS, STAR, a, OPEN, CLOSE,   ɛ;
non terminal e, e_prime, t, t_prime, f;

e ::= e PLUS t | t;
t ::= t STAR f | f;
f ::=   OPEN e CLOSE | a;


terminal a, b, d, ɛ;
non terminal S, A, B, C, D, E;

S ::= a A | b B;
A ::= C a | D b;
B ::= C b | D a;
C ::= E;
D ::= E;
E ::= ɛ;


non terminal S,C;
terminal x,d;

S ::= C C;
C ::= x C | d;

non terminal E, T;
terminal PLUS, ID, NUMBER;

E ::=  T PLUS E | T;
T ::= NUMBER | ID;

non terminal E, T;
terminal PLUS, ID, OPEN, CLOSE;

E ::=  T PLUS E | T;
T ::= ID | OPEN E CLOSE;

non terminal S;
terminal a, b;

S ::= a S b S | a;

non terminal E,T,F;
terminal STAR, PLUS, OPEN, CLOSE, num;

E ::= T PLUS E | T;
T ::= F STAR T | F;
F ::= num | OPEN E CLOSE;

non terminal S;
terminal a, b;

S ::= a S b S | a;

non terminal Integer E, T, F;
non terminal S;
terminal Integer num;
terminal PLUS,MINUS,MUL,DIV,OPEN,CLOSE,ɛ;

S ::= E:e {:System.out.println(e);:};
E ::= E:e {:System.out.println(t);:} PLUS T:t {:RESULT = e + t;:}
     | E:e MINUS T:t {:RESULT = e - t;:}
     | T : t {:RESULT = t;:};
T ::= T:t MUL F:f {:RESULT = t * f;:}
     |T:t DIV F:f {:RESULT = t / f;:}
     |F:f {:RESULT = f;:};
F ::= num:n{:RESULT = n;:}
     |OPEN E:e CLOSE {:RESULT = e;:};
*/

terminal           SEMI, PLUS, MINUS, MUL, DIV, OPEN, CLOSE,ɛ;
terminal Integer   NUMBER;
terminal           EQUALS;
terminal String    ID;
non terminal E, EQUATION_LIST, EQUATION, T, F;

    EQUATION_LIST ::= EQUATION EQUATION_LIST  | ɛ;

    EQUATION ::= E EQUALS E SEMI;

    E ::= T | E MINUS T | E PLUS T;

    T ::= F
        | T DIV F
        | T MUL F;

    F ::= NUMBER
        | OPEN E CLOSE;
